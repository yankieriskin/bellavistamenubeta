async function loadData(){
  try{
    if(window.storage){
      const s=await window.storage.get('menu:settings').catch(()=>null);
      const c=await window.storage.get('menu:categories').catch(()=>null);
      const p=await window.storage.get('menu:products').catch(()=>null);
      if(s&&s.value){try{state.settings={...state.settings,...JSON.parse(s.value)}}catch(e){}}
      if(c&&c.value){try{state.categories=JSON.parse(c.value)}catch(e){state.categories=[]}}
      if(p&&p.value){try{state.products=JSON.parse(p.value)}catch(e){state.products=[]}}
    }else{
      const ls=localStorage.getItem('menu_data_v2');
      if(ls){const d=JSON.parse(ls);state.settings={...state.settings,...(d.settings||{})};state.categories=d.categories||[];state.products=d.products||[]}
    }
    if(state.categories.length===0&&state.products.length===0){seedDefault();await saveAll()}
    // Geriye dönük uyumluluk: parentId/image alanları yoksa ekle
    state.categories.forEach(c=>{if(c.parentId===undefined)c.parentId=null;if(c.image===undefined)c.image=''});
  }catch(err){console.error('Load error:',err);if(state.categories.length===0&&state.products.length===0)seedDefault()}
}

async function saveAll(){
  try{
    if(window.storage){
      await window.storage.set('menu:settings',JSON.stringify(state.settings));
      await window.storage.set('menu:categories',JSON.stringify(state.categories));
      await window.storage.set('menu:products',JSON.stringify(state.products));
    }else{
      localStorage.setItem('menu_data_v2',JSON.stringify({settings:state.settings,categories:state.categories,products:state.products}));
    }
  }catch(err){console.error('Save error:',err);showToast('Kaydetme hatası','error')}
}

function adjustColor(hex,amt){let c=hex.replace('#','');if(c.length===3)c=c.split('').map(x=>x+x).join('');let r=Math.max(0,Math.min(255,parseInt(c.slice(0,2),16)+amt));let g=Math.max(0,Math.min(255,parseInt(c.slice(2,4),16)+amt));let b=Math.max(0,Math.min(255,parseInt(c.slice(4,6),16)+amt));return'#'+[r,g,b].map(x=>x.toString(16).padStart(2,'0')).join('')}

/* ===== HIERARCHY HELPERS ===== */
function getChildren(parentId){return state.categories.filter(c=>(c.parentId||null)===parentId).sort((a,b)=>(a.order||0)-(b.order||0))}
function getCategory(id){return state.categories.find(c=>c.id===id)}
function getAncestors(id){const out=[];let cur=getCategory(id);while(cur){out.unshift(cur);cur=cur.parentId?getCategory(cur.parentId):null}return out}
// Bir kategori "yaprak" mi? (Alt kategorisi yok)
function isLeafCategory(id){return getChildren(id).length===0}
// Bu kategoride veya alt kategorilerinde tüm ürünler
function getProductsRecursive(catId){
  const result=[];
  const stack=[catId];
  while(stack.length){
    const id=stack.pop();
    state.products.filter(p=>p.categoryId===id).forEach(p=>result.push(p));
    getChildren(id).forEach(c=>stack.push(c.id));
  }
  return result;
}
// Bütün soyların id'lerini döndür (kendi dahil)
function getDescendantIds(id){
  const out=[id];const stack=[id];
  while(stack.length){const x=stack.pop();getChildren(x).forEach(c=>{out.push(c.id);stack.push(c.id)})}
  return out;
}

async function saveCategory(){
  const name=document.getElementById('catName').value.trim();
  const icon=document.getElementById('catIcon').value.trim();
  const parentId=document.getElementById('catParent').value||null;
  if(!name){showToast('Kategori adı gerekli','error');return}
  if(state.editingCategoryId){
    const c=state.categories.find(x=>x.id===state.editingCategoryId);
    c.name=name;c.icon=icon;c.parentId=parentId;c.image=state.tempCategoryImage||'';
  }else{
    const siblingMax=getChildren(parentId).reduce((m,c)=>Math.max(m,c.order||0),-1);
    state.categories.push({id:uid(),name,icon,parentId,image:state.tempCategoryImage||'',order:siblingMax+1});
  }
  await saveAll();closeCategoryDrawer();renderCategoriesList();showToast('Kaydedildi');
}

function deleteCategory(id){
  const descendants=getDescendantIds(id);
  const productCount=state.products.filter(p=>descendants.includes(p.categoryId)).length;
  const childCount=descendants.length-1;
  let msg='Bu kategori silinecek.';
  if(childCount>0)msg=`Bu kategori ve ${childCount} alt kategorisi silinecek.`;
  if(productCount>0)msg+=` İçindeki ${productCount} ürün de silinecek!`;
  confirmAction('Kategoriyi Sil',msg,async()=>{
    state.categories=state.categories.filter(c=>!descendants.includes(c.id));
    state.products=state.products.filter(p=>!descendants.includes(p.categoryId));
    await saveAll();renderCategoriesList();showToast('Silindi');
  });
}

async function moveCategory(id,dir){
  const cat=getCategory(id);if(!cat)return;
  const siblings=getChildren(cat.parentId||null);
  const idx=siblings.findIndex(c=>c.id===id);
  if(idx<0)return;const newIdx=idx+dir;if(newIdx<0||newIdx>=siblings.length)return;
  const a=siblings[idx],b=siblings[newIdx];
  const tmp=a.order;a.order=b.order;b.order=tmp;
  await saveAll();renderCategoriesList();
}

function renderProductsList(){
  const filter=document.getElementById('productFilter');
  const currentVal=filter.value;
  let opts='<option value="">Tüm Kategoriler</option>';
  function buildOpts(parentId,depth){
    getChildren(parentId).forEach(c=>{
      opts+=`<option value="${c.id}">${'— '.repeat(depth)}${escapeHtml(c.icon||'')} ${escapeHtml(c.name)}</option>`;
      buildOpts(c.id,depth+1);
    });
  }
  buildOpts(null,0);
  filter.innerHTML=opts;
  filter.value=currentVal;

  const search=(document.getElementById('productSearch').value||'').toLowerCase().trim();
  const catFilter=filter.value;
  let prods=[...state.products];
  if(catFilter){
    const ids=getDescendantIds(catFilter);
    prods=prods.filter(p=>ids.includes(p.categoryId));
  }
  if(search)prods=prods.filter(p=>p.name.toLowerCase().includes(search)||(p.desc&&p.desc.toLowerCase().includes(search)));

  const list=document.getElementById('productsList');
  if(prods.length===0){list.innerHTML=`<div class="empty-state"><div class="empty-state-icon">🍽️</div><h3 class="empty-state-title">Ürün bulunamadı</h3><p>${state.products.length===0?'İlk ürününüzü ekleyin':'Filtreyi değiştirin'}</p></div>`;return}

  list.innerHTML=prods.map(p=>{
    const cat=getCategory(p.categoryId);
    const path=cat?getAncestors(p.categoryId).map(a=>a.name).join(' › '):'(kategori yok)';
    const thumb=p.image?`<img src="${escapeHtml(p.image)}">`:'🍽️';
    return `<div class="item-row" style="${!p.available?'opacity:.6':''}">
      <div class="item-thumb">${thumb}</div>
      <div class="item-info">
        <div class="item-title">${escapeHtml(p.name)} ${p.tag?`<span class="product-tag" style="margin-left:6px">${escapeHtml(p.tag)}</span>`:''} ${!p.available?'<span style="color:var(--danger);font-size:12px;font-weight:600">• Tükendi</span>':''}</div>
        <div class="item-meta">
          <span>${escapeHtml(path)}</span>
          <span style="color:var(--accent-dark);font-weight:700">${fmtPrice(p.price)}</span>
        </div>
      </div>
      <div class="item-actions">
        <button class="btn ${p.available?'btn-secondary':'btn-primary'} btn-sm" onclick="toggleAvailable('${p.id}')">${p.available?'Tükendi':'Stokta'}</button>
        <button class="btn btn-secondary btn-sm" onclick="openProductDrawer('${p.id}')">Düzenle</button>
        <button class="btn btn-danger btn-sm" onclick="deleteProduct('${p.id}')">Sil</button>
      </div>
    </div>`;
  }).join('');
}

async function saveProduct(){
  const name=document.getElementById('prodName').value.trim();
  const categoryId=document.getElementById('prodCategory').value;
  const desc=document.getElementById('prodDesc').value.trim();
  const price=parseFloat(document.getElementById('prodPrice').value)||0;
  const tag=document.getElementById('prodTag').value;
  const available=document.getElementById('prodAvailable').checked;
  if(!name){showToast('Ürün adı gerekli','error');return}
  if(!categoryId){showToast('Kategori seçin','error');return}
  if(state.editingProductId){
    const p=state.products.find(x=>x.id===state.editingProductId);
    p.name=name;p.categoryId=categoryId;p.desc=desc;p.price=price;p.tag=tag;p.available=available;p.image=state.tempProductImage||'';
  }else{
    state.products.push({id:uid(),name,categoryId,desc,price,tag,available,image:state.tempProductImage||''});
  }
  await saveAll();closeProductDrawer();renderProductsList();renderDashboard();showToast('Kaydedildi');
}

function downloadBackup(){
  const data={settings:state.settings,categories:state.categories,products:state.products,exportedAt:new Date().toISOString(),version:2};
  const blob=new Blob([JSON.stringify(data,null,2)],{type:'application/json'});
  const url=URL.createObjectURL(blob);
  const a=document.createElement('a');
  a.href=url;a.download=`menu-yedek-${new Date().toISOString().slice(0,10)}.json`;
  document.body.appendChild(a);a.click();a.remove();URL.revokeObjectURL(url);
  showToast('Yedek indirildi');
}

function restoreBackup(e){
  const f=e.target.files[0];if(!f)return;
  const r=new FileReader();
  r.onload=async ev=>{
    try{
      const d=JSON.parse(ev.target.result);
      if(!d.categories||!d.products)throw new Error('Geçersiz dosya');
      confirmAction('Yedeği Geri Yükle','Mevcut tüm verilerin üzerine yazılacak. Emin misiniz?',async()=>{
        if(d.settings)state.settings={...state.settings,...d.settings};
        state.categories=d.categories||[];
        state.categories.forEach(c=>{if(c.parentId===undefined)c.parentId=null;if(c.image===undefined)c.image=''});
        state.products=d.products||[];
        await saveAll();renderAdmin();showToast('Yedek geri yüklendi');
      });
    }catch(err){showToast('Geçersiz yedek dosyası','error')}
  };
  r.readAsText(f);
  e.target.value='';
}
